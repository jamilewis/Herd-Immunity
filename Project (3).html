Combining Datasets

The population data for each country in 2020 was sourced from a separate dataset and merged with the COVID-19 data to calculate an accurate illness rate. Additionally, the datasets had the 'Country' column labeled with different names, so in order to combine the data based on the country name, I first ensured the 'Country' columns were consistently named. I changed the column name in both datasets to 'Country,' which allowed me to merge them.

def country_info(country_name, df):
    '''takes in a country name and the dataset returns the sic, recovery, and death rates for country'''
    
    # Filter the DataFrame to get the data for the specified country
    country = merged_data[merged_data["Country"] == country_name]

    # Accessing the first row of data for the given country
    country_data = country.iloc[0]

    # Calculate death rate, recovery rate, and sick rate as percentage
    death_rate = country_data["Deaths"] / country_data["Confirmed"] if country_data["Confirmed"] > 0 else 0
    rate_recovery = country_data["Recovered"] / country_data["Confirmed"] if country_data["Confirmed"] > 0 else 0
    sick_rate = (country_data["Active"] / country_data["Population (2020)"]) * 100 if country_data["Population (2020)"] > 0 else 0

    return death_rate, rate_recovery, sick_rate

# Parameters
population_size = 1000
initial_sick = 10
initial_healthy = population_size - initial_sick
initial_dead = 0
infection_rate = sick_rate  
recovery_rate = recovery_rate 
death_rate = death_rate
vaccination_rates = [0.0, 0.3, 0.5, 0.7]  # Different vaccination rates

def simulate_disease(vaccination_rate):
    # Initialize populations
    vaccinated_healthy = int(vaccination_rate * initial_healthy)  # Vaccinated individuals
    healthy = initial_healthy - vaccinated_healthy  # Non-vaccinated healthy individuals
    sick = initial_sick  # Initially sick individuals
    dead = initial_dead  # Initially dead individuals

    # Lists to track population counts over time
    healthy_pop = [healthy]
    sick_pop = [sick]
    dead_pop = [dead]

    # Simulating over 200 days
    for day in range(1, 201):
        # Ensure new infections are limited to the remaining healthy population
        new_infections = min(infection_rate * healthy * sick / population_size, healthy)
        new_recoveries = recovery_rate * sick
        new_deaths = death_rate * sick

        # Update populations
        sick += new_infections - new_recoveries - new_deaths
        dead += new_deaths
        healthy -= new_infections

        # Prevent negative population values
        sick = max(sick, 0)
        dead = min(dead, population_size)  # Ensure dead count doesn't exceed total population
        healthy = max(healthy, 0)

        # Append current values to tracking lists
        healthy_pop.append(healthy)
        sick_pop.append(sick)
        dead_pop.append(dead)

    return healthy_pop, sick_pop, dead_pop


## loop through and make a plot for each vaccination rate 
for vaccination_rate in vaccination_rates:
    # Simulate disease dynamics for the current vaccination rate
    healthy_pop, sick_pop, dead_pop = simulate_disease(vaccination_rate)
    
    # Create a new figure for each vaccination rate
    plt.figure(figsize=(10, 6))
    plt.plot(healthy_pop, label='Healthy')
    plt.plot(sick_pop, label='Sick')
    plt.plot(dead_pop, label='Dead')
    
    # Customize the plot
    plt.xlabel('Days')
    plt.ylabel('Unvaccinated Population')
    plt.title(f'Population Dynamics (Vaccination Rate: {vaccination_rate*100:.0f}%)')
    plt.legend()
    plt.grid(True)
    
    # Display the plot
    plt.show()
